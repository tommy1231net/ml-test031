/*
  TRAINING SET: Chicago Taxi Trips 2022
  Goal: Predict payment_type (Multiclass: Credit Card, Cash, Mobile, Prcard)
  Sampling: 10% (~550k rows total for the selected classes)
*/
SELECT
  CAST(payment_type AS STRING) AS label,
  trip_start_timestamp,
  trip_end_timestamp,
  trip_seconds,
  trip_miles,
  fare,
  tips,
  tolls,
  extras,
  trip_total,
  CAST(pickup_community_area AS STRING) AS pickup_area,
  CAST(dropoff_community_area AS STRING) AS dropoff_area,
  CAST(company AS STRING) AS company_id
FROM
  `bigquery-public-data.chicago_taxi_trips.taxi_trips`
WHERE
  -- Date Range
  trip_start_timestamp >= '2022-01-01' AND trip_start_timestamp < '2023-01-01'
  
  -- Quality Filters
  AND fare > 0 AND trip_miles > 0
  
  -- Class Selection: Focus on the top 4 meaningful payment methods
  AND payment_type IN ('Credit Card', 'Cash', 'Mobile', 'Prcard')
  
  -- 10% Deterministic Sampling
  AND ABS(MOD(FARM_FINGERPRINT(CAST(trip_start_timestamp AS STRING)), 1000)) < 100
