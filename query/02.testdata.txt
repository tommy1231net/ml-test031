/*
  Objective: Extract evaluation (test) data for out-of-time validation.
  Sampling: Aligned with the training sampling rate for consistency.
  Time Range: 2023-12-01 to 2023-12-31.
*/
SELECT
  CAST(payment_type AS STRING) AS label,
  pickup_datetime,
  dropoff_datetime,
  passenger_count,
  trip_distance,
  fare_amount,
  extra,
  mta_tax,
  tolls_amount,
  improvement_surcharge,
  total_amount,
  congestion_surcharge,
  CAST(vendor_id AS STRING) AS vendor_id,
  CAST(rate_code_id AS STRING) AS rate_code_id,
  CAST(PULocationID AS STRING) AS pickup_location_id,
  CAST(DOLocationID AS STRING) AS dropoff_location_id
FROM
  `bigquery-public-data.new_york_taxi_trips.tlc_yellow_trips_2023`
WHERE
  -- Date filtering for Hold-out Evaluation period
  pickup_datetime >= '2023-12-01' AND pickup_datetime < '2024-01-01'
  
  -- Consistency in data cleaning logic
  AND fare_amount BETWEEN 0 AND 500
  AND trip_distance > 0 AND trip_distance < 100
  AND passenger_count IS NOT NULL
  AND payment_type IN (1, 2, 3, 4)
  
  -- Maintaining the same sampling ratio
  AND ABS(MOD(FARM_FINGERPRINT(CAST(pickup_datetime AS STRING)), 1000)) < 30
