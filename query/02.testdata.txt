/*
  EVALUATION SET: Chicago Taxi Trips 2023
  Goal: Out-of-Time testing to evaluate model generalization on the following year.
*/
SELECT
  CAST(payment_type AS STRING) AS label,
  trip_start_timestamp,
  trip_end_timestamp,
  trip_seconds,
  trip_miles,
  fare,
  tips,
  tolls,
  extras,
  trip_total,
  CAST(pickup_community_area AS STRING) AS pickup_area,
  CAST(dropoff_community_area AS STRING) AS dropoff_area,
  CAST(company AS STRING) AS company_id
FROM
  `bigquery-public-data.chicago_taxi_trips.taxi_trips`
WHERE
  trip_start_timestamp >= '2023-01-01' AND trip_start_timestamp < '2024-01-01'
  AND fare > 0 AND trip_miles > 0
  AND payment_type IN ('Credit Card', 'Cash', 'Mobile', 'Prcard')
  AND ABS(MOD(FARM_FINGERPRINT(CAST(trip_start_timestamp AS STRING)), 1000)) < 100
